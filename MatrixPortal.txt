  uint8_t b_scaled = (b * appliedBrightness) / 255;

}

void initEffects() {
  for (int i = 0; i < 70; i++) {
    flakes[i].x = random(PANEL_RES_X);
    flakes[i].y = random(64);
    flakes[i].speed = random(5, 25) / 10.0;
  }
  for (int i = 0; i < PANEL_RES_X; i++) matrixDrops[i] = random(-100, 0);
  for (int i = 0; i < 120; i++) {
    stars[i].x = random(-128, 128);
    stars[i].y = random(-32, 32);
    stars[i].z = random(1, 64);
    stars[i].speed = random(1, 3) / 10.0;
  }
  for (int i = 0; i < 90; i++) sparks[i].age = 999;
  for (int i = 0; i < 10; i++) {
    balls[i].x = random(10, PANEL_RES_X - 10);
    balls[i].y = random(10, 54);
    balls[i].vx = (random(0, 2) == 0 ? 1 : -1) * (random(10, 25) / 10.0);
    balls[i].vy = (random(0, 2) == 0 ? 1 : -1) * (random(10, 25) / 10.0);
    balls[i].r = random(3, 7);
    balls[i].color = matrix.color565(random(100, 255), random(100, 255), random(100, 255));
  }
}

void updateSnow() {
  for (int i = 0; i < 70; i++) {
    matrix.drawPixel(flakes[i].x, flakes[i].y, getColor(200, 200, 200));

      flakes[i].x = random(PANEL_RES_X);

}
void updateMatrix() {
  for (int i = 0; i < PANEL_RES_X; i += 2) {
    if (matrixDrops[i] >= 0 && matrixDrops[i] < PANEL_RES_Y) {
      matrix.drawPixel(i, matrixDrops[i], getColor(0, 255, 0));
      if (matrixDrops[i] > 0) matrix.drawPixel(i, matrixDrops[i] - 1, getColor(0, 150, 0));
      if (matrixDrops[i] > 1) matrix.drawPixel(i, matrixDrops[i] - 2, getColor(0, 50, 0));
    }
    matrixDrops[i]++;
    if (matrixDrops[i] > PANEL_RES_Y + random(0, 50)) matrixDrops[i] = random(-20, 0);
  }
}

void updateStars() {
  for (int i = 0; i < 120; i++) {
    stars[i].z -= stars[i].speed;
    if (stars[i].z <= 0) {
      stars[i].x = random(-128, 128);
      stars[i].y = random(-32, 32);
      stars[i].z = 64;
    }
    int sx = (stars[i].x / stars[i].z * 30) + (PANEL_RES_X / 2);
    int sy = (stars[i].y / stars[i].z * 30) + 32;
    if (sx >= 0 && sx < PANEL_RES_X && sy >= 0 && sy < 64) {
      int bri = 255 - (stars[i].z * 4);
      if (bri < 0) bri = 0;
      matrix.drawPixel(sx, sy, getColor(bri, bri, bri));
    }
  }
}
void initFireworks() {
  int cx = random(30, PANEL_RES_X - 30), cy = random(10, 30);
  uint16_t col = matrix.color565(random(100, 255), random(100, 255), random(100, 255));
  for (int i = 0; i < 90; i++) {
    sparks[i].x = cx;
    sparks[i].y = cy;
    float angle = random(0, 360) * 3.1415 / 180.0;
    float speed = random(10, 35) / 10.0;
    sparks[i].vx = cos(angle) * speed;
    sparks[i].vy = sin(angle) * speed;
    sparks[i].age = 0;
    sparks[i].max_age = random(15, 35);
    sparks[i].color = col;
  }
}
void updateFireworks() {
  bool active = false;
  for (int i = 0; i < 90; i++) {
    if (sparks[i].age < sparks[i].max_age) {
      active = true;
      matrix.drawPixel(sparks[i].x, sparks[i].y, sparks[i].color);
      sparks[i].x += sparks[i].vx;
      sparks[i].y += sparks[i].vy;
      sparks[i].vy += 0.05;
      sparks[i].age++;
    }
  }
  if (!active && random(0, 100) < 3) initFireworks();
}
void update3DCube() {
  cubeAngleX += 0.03;
  cubeAngleY += 0.02;
  cubeAngleZ += 0.01;
  int projected[8][2];
  uint8_t r = (sin(millis() / 500.0) * 127) + 128, g = (sin(millis() / 700.0) * 127) + 128, b = (sin(millis() / 900.0) * 127) + 128;
  uint16_t cubeColor = getColor(r, g, b);
  for (int i = 0; i < 8; i++) {
    float x = cubeVertices[i][0], y = cubeVertices[i][1], z = cubeVertices[i][2];
    float xy = cos(cubeAngleX) * y - sin(cubeAngleX) * z, xz = sin(cubeAngleX) * y + cos(cubeAngleX) * z;
    y = xy;
    z = xz;
    float yx = cos(cubeAngleY) * x + sin(cubeAngleY) * z, yz = -sin(cubeAngleY) * x + cos(cubeAngleY) * z;
    x = yx;
    z = yz;
    float zx = cos(cubeAngleZ) * x - sin(cubeAngleZ) * y, zy = sin(cubeAngleZ) * x + cos(cubeAngleZ) * y;
    x = zx;
    y = zy;
    float dist = 2.8;
    projected[i][0] = (x * 35 / (z + dist)) + (PANEL_RES_X / 2);
    projected[i][1] = (y * 35 / (z + dist)) + 32;
  }
  for (int i = 0; i < 12; i++) { matrix.drawLine(projected[cubeEdges[i][0]][0], projected[cubeEdges[i][0]][1], projected[cubeEdges[i][1]][0], projected[cubeEdges[i][1]][1], cubeColor); }
}
void update3DPyramid() {
  pyrAngleX += 0.03;
  pyrAngleY += 0.02;
  pyrAngleZ += 0.01;
  int projected[5][2];
  uint8_t r = (sin(millis() / 400.0) * 127) + 128, g = (sin(millis() / 600.0) * 127) + 128, b = (sin(millis() / 800.0) * 127) + 128;
  uint16_t pyrColor = getColor(r, g, b);
  for (int i = 0; i < 5; i++) {
    float x = pyrVerts[i][0], y = pyrVerts[i][1], z = pyrVerts[i][2];
    float xy = cos(pyrAngleX) * y - sin(pyrAngleX) * z, xz = sin(pyrAngleX) * y + cos(pyrAngleX) * z;
    y = xy;
    z = xz;
    float yx = cos(pyrAngleY) * x + sin(pyrAngleY) * z, yz = -sin(pyrAngleY) * x + cos(pyrAngleY) * z;
    x = yx;
    z = yz;
    float zx = cos(pyrAngleZ) * x - sin(pyrAngleZ) * y, zy = sin(pyrAngleZ) * x + cos(pyrAngleZ) * y;
    x = zx;
    y = zy;
    float dist = 2.8;
    projected[i][0] = (x * 40 / (z + dist)) + (PANEL_RES_X / 2);
    projected[i][1] = (y * 40 / (z + dist)) + 32;
  }
  for (int i = 0; i < 8; i++) { matrix.drawLine(projected[pyrEdges[i][0]][0], projected[pyrEdges[i][0]][1], projected[pyrEdges[i][1]][0], projected[pyrEdges[i][1]][1], pyrColor); }
}

void updateSand() {
  if (!accelFound) return;
  sensors_event_t event;
  lis.getEvent(&event);
  sandSim.iterate(event.acceleration.x * 10, -event.acceleration.y * 10, 0);
  uint16_t sandColor = getColor(255, 200, 50);
  for (int i = 0; i < 2000; i++) {
    uint16_t x, y;
    sandSim.getPosition(i, &x, &y);
    matrix.drawPixel(x, y, sandColor);
  }
}

void updateBalls() {
  for (int i = 0; i < 10; i++) {
    balls[i].x += balls[i].vx;
    balls[i].y += balls[i].vy;
    if (balls[i].x - balls[i].r <= 0 || balls[i].x + balls[i].r >= PANEL_RES_X - 1) balls[i].vx *= -1;
    if (balls[i].y - balls[i].r <= 0 || balls[i].y + balls[i].r >= 63) balls[i].vy *= -1;
    matrix.fillCircle(balls[i].x, balls[i].y, balls[i].r, balls[i].color);
  }
}
void updateWaves() {
  waveOffset1 += 0.05;
  waveOffset2 -= 0.03;
  for (int x = 0; x < PANEL_RES_X; x += 3) {
    int y1 = 32 + sin(x * 0.05 + waveOffset1) * 16;
    int y2 = 32 + cos(x * 0.08 + waveOffset2) * 12;
    matrix.fillCircle(x, y1, 1, getColor(0, 255, 255));
    matrix.fillCircle(x, y2, 1, getColor(255, 0, 255));
  }
}

uint16_t hexToRGB565(String hex) {
  if (hex.startsWith("#")) hex.remove(0, 1);
  long rgb = strtol(hex.c_str(), NULL, 16);
  return getColor((rgb >> 16) & 0xFF, (rgb >> 8) & 0xFF, rgb & 0xFF);
}

// ================= ЧТЕНИЕ SD КАРТЫ =================
uint16_t read16(File &f) {
  uint16_t result;
  ((uint8_t *)&result)[0] = f.read();
  ((uint8_t *)&result)[1] = f.read();
  return result;
}
uint32_t read32(File &f) {
  uint32_t result;
  ((uint8_t *)&result)[0] = f.read();
  ((uint8_t *)&result)[1] = f.read();
  ((uint8_t *)&result)[2] = f.read();
  ((uint8_t *)&result)[3] = f.read();
  return result;
}

void drawBmp(const char *filename, int16_t x, int16_t y) {
  if ((x >= PANEL_RES_X) || (y >= 64) || (x + 128 < 0)) return;
  File bmpFile = SD.open(filename);
  if (!bmpFile) return;
  uint32_t seekOffset, w, h;
  if (read16(bmpFile) == 0x4D42) {
    read32(bmpFile);
    read32(bmpFile);
    seekOffset = read32(bmpFile);
    read32(bmpFile);
    w = read32(bmpFile);
    h = read32(bmpFile);
    if (read16(bmpFile) == 1 && read16(bmpFile) == 24 && read32(bmpFile) == 0) {
      int rowSize = (w * 3 + 3) & ~3;
      uint8_t lineBuffer[rowSize];
      for (int row = 0; row < h; row++) {
        int drawY = y + h - 1 - row;
        if (drawY < 0 || drawY >= 64) continue;
        bmpFile.seek(seekOffset + row * rowSize);
        bmpFile.read(lineBuffer, rowSize);
        for (int col = 0; col < w; col++) {
          int drawX = x + col;
          if (drawX < 0 || drawX >= PANEL_RES_X) continue;
          uint8_t b = lineBuffer[col * 3], g = lineBuffer[col * 3 + 1], r = lineBuffer[col * 3 + 2];
          if (r > 5 || g > 5 || b > 5) matrix.drawPixel(drawX, drawY, getColor(r, g, b));
        }
      }
    }
  }
  bmpFile.close();
}

String cp1251_to_utf8(String cp1251) {
  String res = "";
  res.reserve(cp1251.length() * 2);
  for (int i = 0; i < cp1251.length(); i++) {
    uint8_t c = cp1251[i];
    if (c < 128) res += (char)c;
    else if (c >= 192 && c <= 239) {
      res += (char)0xD0;
      res += (char)(c - 48);
    } else if (c >= 240 && c <= 255) {
      res += (char)0xD1;
      res += (char)(c - 112);
    } else if (c == 168) {
      res += (char)0xD0;
      res += (char)0x81;
    } else if (c == 184) {
      res += (char)0xD1;
      res += (char)0x91;
    } else {
      res += ' ';
    }
  }
  return res;
}
int getMixedTextWidth(String text) {
  int w = 0, i = 0;
  while (i < text.length()) {
    int tagStart = text.indexOf('[', i);
    if (tagStart == -1) {
      w += u8g2_fonts.getUTF8Width(text.substring(i).c_str());
      break;
    }
    w += u8g2_fonts.getUTF8Width(text.substring(i, tagStart).c_str());
    int tagEnd = text.indexOf(']', tagStart);
    if (tagEnd != -1) {
      w += 26;
      i = tagEnd + 1;
    } else {
      w += u8g2_fonts.getUTF8Width("[");
      i = tagStart + 1;
    }
  }
  return w;
}
void drawMixedText(String text, int x, int y) {
  int currX = x, i = 0;
  while (i < text.length()) {
    int tagStart = text.indexOf('[', i);
    if (tagStart == -1) {
      u8g2_fonts.setCursor(currX, y);
      u8g2_fonts.print(text.substring(i));
      break;
    }
    String sub = text.substring(i, tagStart);
    u8g2_fonts.setCursor(currX, y);
    u8g2_fonts.print(sub);
    currX += u8g2_fonts.getUTF8Width(sub.c_str());
    int tagEnd = text.indexOf(']', tagStart);
    if (tagEnd != -1) {
      String tag = text.substring(tagStart + 1, tagEnd);
      String filename = "/" + tag + ".bmp";
      drawBmp(filename.c_str(), currX, y - 24);
      currX += 26;
      i = tagEnd + 1;
    } else {
      u8g2_fonts.setCursor(currX, y);
      u8g2_fonts.print("[");
      currX += u8g2_fonts.getUTF8Width("[");
      i = tagStart + 1;
    }
  }
}

void calculateCountdown() {
  struct tm targetInfo = { 0 };
  int yr, mo, dy;
  if (sscanf(countdownTargetDate.c_str(), "%d-%d-%d", &yr, &mo, &dy) == 3) {
    targetInfo.tm_year = yr - 1900;
    targetInfo.tm_mon = mo - 1;
    targetInfo.tm_mday = dy;
    targetInfo.tm_isdst = -1;
    daysRemaining = (mktime(&targetInfo) - time(NULL)) / 86400;
  }
}

void checkBirthdays() {
  if (!globalTimeOk || birthdayList.length() == 0) {
    hasBirthdayTomorrow = false;
    hasBirthdayToday = false;
    return;
  }
  time_t now = time(NULL);
  struct tm *tm_today = localtime(&now);
  char todayStr[10];
  sprintf(todayStr, "%02d.%02d", tm_today->tm_mday, tm_today->tm_mon + 1);
  String searchToday = String(todayStr);
  time_t tomorrow = now + 86400;
  struct tm *tm_tom = localtime(&tomorrow);
  char tomStr[10];
  sprintf(tomStr, "%02d.%02d", tm_tom->tm_mday, tm_tom->tm_mon + 1);
  String searchTom = String(tomStr);

  hasBirthdayTomorrow = false;
  tomorrowBirthdaysNames = "";
  hasBirthdayToday = false;
  todayBirthdaysNames = "";
  int startIdx = 0;
  while (startIdx < birthdayList.length()) {
    int commaIdx = birthdayList.indexOf(',', startIdx);
    if (commaIdx == -1) commaIdx = birthdayList.length();
    String entry = birthdayList.substring(startIdx, commaIdx);
    entry.trim();
    if (entry.endsWith(searchTom)) {
      int spaceIdx = entry.lastIndexOf(' ');
      String name = (spaceIdx != -1) ? entry.substring(0, spaceIdx) : "Кто-то";
      if (tomorrowBirthdaysNames.length() > 0) tomorrowBirthdaysNames += " и ";
      tomorrowBirthdaysNames += name;
      hasBirthdayTomorrow = true;
    }
    if (entry.endsWith(searchToday)) {
      int spaceIdx = entry.lastIndexOf(' ');
      String name = (spaceIdx != -1) ? entry.substring(0, spaceIdx) : "Кто-то";
      if (todayBirthdaysNames.length() > 0) todayBirthdaysNames += " и ";
      todayBirthdaysNames += name;
      hasBirthdayToday = true;
    }
    startIdx = commaIdx + 1;
  }
  buildScreenSequence();
}

void fetchNews() {
  if (WiFi.status() != WL_CONNECTED) return;
  WiFiClientSecure client;
  client.setInsecure();
  HTTPClient http;
  http.setTimeout(6000);
  http.begin(client, "https://lenta.ru/rss/news");
  if (http.GET() == 200) {
    String payload = http.getString();
    String parsedNews = " 🌍 ГЛАВНЫЕ новости: ";
    int startIndex = payload.indexOf("<title>");
    if (startIndex != -1) startIndex += 7;
    int count = 0;
    while (count < 4) {
      startIndex = payload.indexOf("<title>", startIndex);
      if (startIndex == -1) break;
      startIndex += 7;
      int endIndex = payload.indexOf("</title>", startIndex);
      if (endIndex != -1) {

        title.replace("<![CDATA[", "");
        title.replace("]]>", "");
        parsedNews += " • " + title;
        count++;
      }
    }
    if (count > 0) newsText = parsedNews + "   *** ";
  }
  http.end();
}

void networkTaskCode(void *parameter) {
  for (;;) {
    if (WiFi.status() != WL_CONNECTED) {
      WiFi.disconnect();
      WiFi.begin();
      vTaskDelay(5000 / portTICK_PERIOD_MS);
      continue;
    }
    unsigned long ms = millis();
    if (ms - lastDataUpdate >= (dataLoaded ? 900000 : 10000)) {
      fetchAllData();
      lastDataUpdate = millis();
    }
    if (millis() - lastTgCheck >= 5000) {
      checkTelegram();
      lastTgCheck = millis();
    }
    if (showJokeFlag && needNewJoke) {
      fetchJoke();
      needNewJoke = false;
    }
    vTaskDelay(100 / portTICK_PERIOD_MS);
  }
}

void drawWiFiIcon(int x, int y) {
  uint16_t color = (WiFi.RSSI() < -75) ? getColor(255, 0, 0) : getColor(0, 255, 0);
  int step = (millis() / 300) % 4;
  matrix.drawPixel(x + 2, y + 4, color);
  if (step > 0) matrix.drawLine(x + 1, y + 3, x + 3, y + 3, color);
  if (step > 1) matrix.drawLine(x, y + 2, x + 4, y + 2, color);
}

void drawAnimatedWeather(int cx, int cy, String code) {
  unsigned long t = millis();
  uint16_t cYellow = getColor(255, 200, 0), cWhite = getColor(255, 255, 255), cLightGray = getColor(180, 180, 180);
  uint16_t cBlue = getColor(50, 150, 255), cGray = getColor(100, 100, 100), cBlack = getColor(0, 0, 0);

  if (code == "01d") {
    matrix.fillCircle(cx, cy, 11, cYellow);
    for (int i = 0; i < 8; i++) {
      float angle = (t / 800.0) + (i * PI / 4.0);
      matrix.drawLine(cx + cos(angle) * 14, cy + sin(angle) * 14, cx + cos(angle) * 20, cy + sin(angle) * 20, cYellow);
    }
  } else if (code == "01n") {
    matrix.fillCircle(cx, cy, 12, cYellow);
    matrix.fillCircle(cx - 4, cy - 4, 10, cBlack);
  } else if (code.indexOf("02") != -1 || code.indexOf("03") != -1 || code.indexOf("04") != -1 || code.indexOf("50") != -1) {
    int dy = sin(t / 400.0) * 3;
    if (code == "02d") { matrix.fillCircle(cx + 8, cy - 6 + dy, 9, cYellow); }
    if (code == "02n") {
      matrix.fillCircle(cx + 8, cy - 6 + dy, 9, cYellow);
      matrix.fillCircle(cx + 4, cy - 10 + dy, 7, cBlack);
    }
    matrix.fillCircle(cx - 7, cy + 7 + dy, 9, cLightGray);
    matrix.fillCircle(cx + 7, cy + 1 + dy, 12, cLightGray);
    matrix.fillCircle(cx + 17, cy + 7 + dy, 9, cLightGray);
    matrix.fillRect(cx - 7, cy + 1 + dy, 24, 16, cLightGray);
    matrix.fillCircle(cx - 5, cy + 5 + dy, 9, cWhite);
    matrix.fillCircle(cx + 5, cy - 1 + dy, 12, cWhite);
    matrix.fillCircle(cx + 15, cy + 5 + dy, 9, cWhite);
    matrix.fillRect(cx - 5, cy - 1 + dy, 20, 16, cWhite);
  } else if (code.indexOf("09") != -1 || code.indexOf("10") != -1) {
    matrix.fillCircle(cx - 5, cy + 1, 9, cGray);
    matrix.fillCircle(cx + 5, cy - 5, 12, cGray);
    matrix.fillCircle(cx + 15, cy + 1, 9, cGray);
    matrix.fillRect(cx - 5, cy - 5, 20, 16, cGray);
    for (int i = 0; i < 4; i++) {
      int dropY = ((t / (20 + i * 5)) + i * 12) % 20;
      int dropX = cx - 8 + i * 8;
      matrix.drawLine(dropX, cy + 10 + dropY, dropX - 2, cy + 14 + dropY, cBlue);
    }
  } else if (code.indexOf("13") != -1) {
    matrix.fillCircle(cx - 5, cy + 1, 9, cWhite);
    matrix.fillCircle(cx + 5, cy - 5, 12, cWhite);
    matrix.fillCircle(cx + 15, cy + 1, 9, cWhite);
    matrix.fillRect(cx - 5, cy - 5, 20, 16, cWhite);
    for (int i = 0; i < 4; i++) {
      int snowY = ((t / (40 + i * 10)) + i * 10) % 20;
      int snowX = cx - 8 + i * 8 + sin((t / 300.0) + i) * 4;
      matrix.drawPixel(snowX, cy + 10 + snowY, cWhite);
      matrix.drawPixel(snowX + 1, cy + 10 + snowY, cWhite);
    }
  } else if (code.indexOf("11") != -1) {
    matrix.fillCircle(cx - 5, cy + 1, 9, cGray);
    matrix.fillCircle(cx + 5, cy - 5, 12, cGray);
    matrix.fillCircle(cx + 15, cy + 1, 9, cGray);
    matrix.fillRect(cx - 5, cy - 5, 20, 16, cGray);
    if ((t % 1500) < 200) {
      matrix.drawLine(cx + 5, cy + 7, cx, cy + 16, cYellow);
      matrix.drawLine(cx, cy + 16, cx + 8, cy + 16, cYellow);
      matrix.drawLine(cx + 8, cy + 16, cx + 2, cy + 28, cYellow);
      if ((t % 1500) < 50) matrix.fillCircle(cx + 5, cy - 5, 10, cWhite);
    }
  }
}

void drawBTTFRow(int ox, int y, uint16_t color, String title, String mon, int day, int year, int h, int m, bool isPresent) {
  uint16_t titleColor = getColor(255, 255, 255);
  u8g2_fonts.setFont(u8g2_font_5x7_tr);
  u8g2_fonts.setForegroundColor(titleColor);
  int tw = u8g2_fonts.getUTF8Width(title.c_str());
  matrix.fillRect(ox + (PANEL_RES_X - tw) / 2 - 2, y - 11, tw + 4, 9, getColor(100, 0, 0));
  u8g2_fonts.setCursor(ox + (PANEL_RES_X - tw) / 2, y - 3);
  u8g2_fonts.print(title);
  u8g2_fonts.setFont(u8g2_font_helvB12_te);
  u8g2_fonts.setForegroundColor(color);
  int vy = y + 10;
  u8g2_fonts.setCursor(ox + 16, vy);
  u8g2_fonts.print(mon);
  u8g2_fonts.setCursor(ox + 72, vy);
  u8g2_fonts.printf("%02d", day);
  u8g2_fonts.setCursor(ox + 114, vy);
  u8g2_fonts.printf("%04d", year);
  u8g2_fonts.setCursor(ox + 182, vy);
  u8g2_fonts.printf("%02d", h);
  u8g2_fonts.setCursor(ox + 224, vy);
  u8g2_fonts.printf("%02d", m);
  if (!isPresent || (millis() / 500) % 2 == 0) {
    matrix.fillCircle(ox + 214, vy - 9, 1, color);
    matrix.fillCircle(ox + 214, vy - 3, 1, color);
  }
}

        if (bdayNameScrollX < -fullWidth) bdayNameScrollX = PANEL_RES_X;

    } else {

void updateTerminatorLog() {
  unsigned long ms = millis();
  int scrollDelay = 80;
  if (ms - lastTermScrollTime > scrollDelay) {
    termScrollY += 1.0;
    lastTermScrollTime = ms;
    if (termScrollY >= 9.0) {
      termScrollY = 0;
      for (int i = TERM_LINES - 1; i > 0; i--) { termLog[i] = termLog[i - 1]; }
      char timeStr[16] = "00:00:00", dateStr[16] = "00.00.0000";
      if (globalTimeOk) {
        strftime(timeStr, sizeof(timeStr), "%H:%M:%S", &globalTimeInfo);
        strftime(dateStr, sizeof(dateStr), "%d.%m.%Y", &globalTimeInfo);
      }
      termStep++;
      int stepType = termStep % 7;
      if (stepType == 0) termLog[0] = String(timeStr) + " RAM " + String(ESP.getFreeHeap() / 1024) + "K";
      else if (stepType == 1) termLog[0] = "WIFI: " + String(WiFi.RSSI()) + " dBm";
      else if (stepType == 2) termLog[0] = "DATE: " + String(dateStr);
      else if (stepType == 3) termLog[0] = String(timeStr) + " SYS OK";
      else if (stepType == 4) termLog[0] = "THREAT: LOW";
      else if (stepType == 5) termLog[0] = "SCAN: ACTIVE";
      else {
        String rHex = "0x" + String(random(0x1000, 0xFFFF), HEX);
        rHex.toUpperCase();
        termLog[0] = "MEM: " + rHex;
      }
    }
  }
}

void drawSleepyEyes(bool isSurprised) {
  unsigned long ms = millis();
  if (eyeType == 5) {
    uint16_t termRed = getColor(255, 0, 0), termDimRed = getColor(180, 0, 0), termDarkRed = getColor(100, 0, 0), termWhite = getColor(255, 255, 255);
    int cx = 64, cy = 32;
    float zoom = isSurprised ? 0.6 : (1.0 + 0.15 * sin(ms / 400.0));
    matrix.drawCircle(cx, cy, 26 * zoom, termDarkRed);
    matrix.drawCircle(cx, cy, 25 * zoom, termDarkRed);
    matrix.drawCircle(cx, cy, 18 * zoom, termRed);
    matrix.drawCircle(cx, cy, 10 * zoom, termRed);
    matrix.fillCircle(cx, cy, 5 * zoom, isSurprised ? termWhite : termRed);
    matrix.drawLine(cx - 30, cy, cx - 12, cy, termDarkRed);
    matrix.drawLine(cx + 12, cy, cx + 30, cy, termDarkRed);
    matrix.drawLine(cx, cy - 30, cx, cy - 12, termDarkRed);
    matrix.drawLine(cx, cy + 12, cx, cy + 30, termDarkRed);
    if (isSurprised) {
      u8g2_fonts.setFont(u8g2_font_cu12_t_cyrillic);
      u8g2_fonts.setForegroundColor(termRed);
      if ((ms / 200) % 2 == 0) {
        u8g2_fonts.setCursor(140, 28);
        u8g2_fonts.print("WARNING:");
        u8g2_fonts.setCursor(130, 48);
        u8g2_fonts.print("INCOMING DATA");
      }
    } else {
      u8g2_fonts.setFont(u8g2_font_5x7_tr);
      for (int i = 0; i < TERM_LINES; i++) {
        if (termLog[i].length() > 0) {
          if (i <= 2) u8g2_fonts.setForegroundColor(termRed);
          else if (i <= 5) u8g2_fonts.setForegroundColor(termDimRed);
          else u8g2_fonts.setForegroundColor(termDarkRed);
          int yPos = (i * 9) + (int)termScrollY + 7;
          if (yPos > 6 && yPos <= 74) {
            u8g2_fonts.setCursor(130, yPos);
            u8g2_fonts.print(termLog[i]);
          }
        }
      }
    }
    int scanSpeed = isSurprised ? 8 : 25;
    int scanY = (ms / scanSpeed) % PANEL_RES_Y;
    matrix.drawLine(0, scanY, PANEL_RES_X, scanY, termRed);
    matrix.drawLine(0, scanY - 1, PANEL_RES_X, scanY - 1, termDarkRed);
    return;
  }

  static unsigned long eyeLastBlink = 0, eyeLastLook = 0;
  static float eyeLookTargetX = 0, eyeLookTargetY = 0, eyeLookCurrentX = 0, eyeLookCurrentY = 0;
  static bool eyeIsBlinking = false;
  static int eyeBlinkDuration = 300;
  static float blinkDepth = 1.0;
  if (isSurprised) {
    eyeLookTargetX = 0;
    eyeLookTargetY = 0;
    eyeIsBlinking = false;
  } else {
    if (!eyeIsBlinking && ms - eyeLastBlink > random(5000, 15000)) {
      eyeIsBlinking = true;
      eyeLastBlink = ms;
      if (random(0, 100) < 30) {
        blinkDepth = (float)random(30, 60) / 100.0;
        eyeBlinkDuration = random(150, 250);
      } else {
        blinkDepth = 1.0;
        eyeBlinkDuration = random(300, 450);
      }
    }
    if (ms - eyeLastLook > random(4000, 8000)) {
      eyeLastLook = ms;
      if (random(0, 3) == 0) {
        eyeLookTargetX = 0;
        eyeLookTargetY = 0;
      } else {
        eyeLookTargetX = random(-14, 15);
        eyeLookTargetY = random(-5, 6);
      }
    }
  }
  eyeLookCurrentX += (eyeLookTargetX - eyeLookCurrentX) * 0.08;
  eyeLookCurrentY += (eyeLookTargetY - eyeLookCurrentY) * 0.08;
  float blinkP = 0;
  if (eyeIsBlinking && !isSurprised) {
    long t = ms - eyeLastBlink;
    if (t > eyeBlinkDuration) {
      eyeIsBlinking = false;
      eyeLastBlink = ms;
    } else {
      if (t < eyeBlinkDuration / 2) blinkP = (float)t / (eyeBlinkDuration / 2);
      else blinkP = 1.0 - (float)(t - eyeBlinkDuration / 2) / (eyeBlinkDuration / 2);
      blinkP *= blinkDepth;
    }
  }
  uint16_t eColor = hexToRGB565(eyeColorHex);
  int cxL = PANEL_RES_X / 4, cxR = PANEL_RES_X * 3 / 4, cy = PANEL_RES_Y / 2, r = 26;

  for (int i = 0; i < 2; i++) {
    int cx = (i == 0) ? cxL : cxR;

    if (eyeType == 0 || eyeType == 1) {
      matrix.fillCircle(cx, cy, r, eColor);
      if (eyeType == 0) {
        int pR = isSurprised ? 6 : 11;
        matrix.fillCircle(cx + (int)eyeLookCurrentX, cy + (int)eyeLookCurrentY, pR, 0);
      } else {
        int pW = isSurprised ? 4 : 10, pH = isSurprised ? 20 : 32;
        matrix.fillRoundRect(cx + (int)eyeLookCurrentX - pW / 2, cy + (int)eyeLookCurrentY - pH / 2, pW, pH, pW / 2, 0);
      }
    } else if (eyeType == 6) {
      int eyeW = 70, eyeH = 34;
      int ex = cx - eyeW / 2;
      int ey = cy - eyeH / 2 + 4;
      matrix.fillRoundRect(ex, ey, eyeW, eyeH, 15, eColor);
      int squint = isSurprised ? 0 : 10;
      matrix.fillRect(ex, ey, eyeW, squint, 0);
      int pW = isSurprised ? 8 : 12;
      int pH = isSurprised ? 22 : 20;
      int px = cx + (int)eyeLookCurrentX;
      int py = cy + (int)eyeLookCurrentY + 4;
      matrix.fillRoundRect(px - pW / 2, py - pH / 2 + (squint / 2), pW, pH, pW / 2, 0);
      if (!isSurprised) {
        matrix.fillCircle(px + 4, py - 4, 2, getColor(255, 255, 255));
        matrix.fillCircle(px - 2, py + 3, 1, getColor(255, 255, 255));
      }
    } else if (eyeType == 2) {
      matrix.drawCircle(cx, cy, r, eColor);
      matrix.drawCircle(cx, cy, r - 1, eColor);
      matrix.drawLine(cx - r, cy, cx + r, cy, eColor);
      matrix.drawLine(cx, cy - r, cx, cy + r, eColor);
      uint16_t coreC = isSurprised ? getColor(255, 0, 0) : getColor(255, 255, 255);
      matrix.fillRect(cx + (int)eyeLookCurrentX - 5, cy + (int)eyeLookCurrentY - 5, 10, 10, coreC);
    } else if (eyeType == 3) {
      matrix.fillCircle(cx, cy, r, eColor);
      int px = cx + (int)eyeLookCurrentX, py = cy + (int)eyeLookCurrentY, maxR = isSurprised ? 20 : 16;
      for (int rad = 3; rad <= maxR; rad += 5) {
        matrix.drawCircle(px, py, rad, 0);
        matrix.drawCircle(px, py, rad + 1, 0);
      }
    } else if (eyeType == 4) {
      int sqR = 24;
      matrix.fillRect(cx - sqR, cy - sqR, sqR * 2, sqR * 2, eColor);
      int pW = isSurprised ? 6 : 16;
      matrix.fillRect(cx + (int)eyeLookCurrentX - pW / 2, cy + (int)eyeLookCurrentY - pW / 2, pW, pW, 0);
    }

    if (eyeType == 0 || eyeType == 1 || eyeType == 6) {
      int glintX = cx + 5 + (int)(eyeLookCurrentX * 0.5);
      int glintY = cy - 8 + (int)(eyeLookCurrentY * 0.5);
      matrix.fillCircle(glintX, glintY, 2, getColor(255, 255, 255));
    }

    if (!isSurprised) {
      if (eyeType == 4) {
        int drop = 14 + (blinkP * (r + 4));
        matrix.fillRect(cx - 26, cy - 26, 52, drop, 0);
      } else if (eyeType == 6) {
        int foxDrop = blinkP * 40;
        matrix.fillRect(cx - 40, cy - 25, 80, foxDrop, 0);
      } else {
        int drop = 14 + (blinkP * (r + 4));
        matrix.fillRect(cx - r - 2, cy - r - 2, (r * 2) + 4, drop, 0);
      }
    }
  }
}

              matrix.fillRect(0, 0, PANEL_RES_X, wy, 0);
              for (int i = 0; i < PANEL_RES_X; i += random(2, 6)) {
                if (wy + random(-10, 5) < PANEL_RES_Y) matrix.drawLine(i, wy, i, wy - random(5, 15), getColor(0, 255, 0));
              }
            }
          } else if (activeAnimType == 13) {
            if (p < 0.5) {
              int offset = (p * 2.0) * 15.0;
              drawScreen(activeScreens[currentScreenIdx], random(-offset, offset), random(-offset, offset), &globalTimeInfo, globalTimeOk);
            } else {
              int offset = ((1.0 - p) * 2.0) * 15.0;
              drawScreen(activeScreens[nextScreenIdx], random(-offset, offset), random(-offset, offset), &globalTimeInfo, globalTimeOk);
            }
          } else if (activeAnimType == 14) {
            if (p < 0.5) {
              drawScreen(activeScreens[currentScreenIdx], 0, 0, &globalTimeInfo, globalTimeOk);
              for (int x = 0; x < PANEL_RES_X; x++) {
                int h = (p * 2.0) * PANEL_RES_Y + random(-15, 15);
                if (h > 0) matrix.drawLine(x, 0, x, h, 0);
              }
            } else {
              drawScreen(activeScreens[nextScreenIdx], 0, 0, &globalTimeInfo, globalTimeOk);
              for (int x = 0; x < PANEL_RES_X; x++) {
                int h = ((1.0 - p) * 2.0) * PANEL_RES_Y + random(-15, 15);
                if (h > 0) matrix.drawLine(x, PANEL_RES_Y - h, x, PANEL_RES_Y, 0);
              }
            }
          } else if (activeAnimType == 15) {
            if (p < 0.3) {
              drawScreen(activeScreens[currentScreenIdx], 0, 0, &globalTimeInfo, globalTimeOk);
              int h = (p / 0.3) * (PANEL_RES_Y / 2);
              matrix.fillRect(0, 0, PANEL_RES_X, h, 0);
              matrix.fillRect(0, PANEL_RES_Y - h, PANEL_RES_X, h, 0);
            } else if (p < 0.4) {
              matrix.drawLine(0, PANEL_RES_Y / 2, PANEL_RES_X, PANEL_RES_Y / 2, getColor(255, 255, 255));
            } else if (p < 0.6) {
              matrix.fillCircle(PANEL_RES_X / 2, PANEL_RES_Y / 2, 2, getColor(255, 255, 255));
            } else if (p < 0.7) {
              matrix.drawLine(0, PANEL_RES_Y / 2, PANEL_RES_X, PANEL_RES_Y / 2, getColor(255, 255, 255));
            } else {
              drawScreen(activeScreens[nextScreenIdx], 0, 0, &globalTimeInfo, globalTimeOk);
              int h = ((1.0 - p) / 0.3) * (PANEL_RES_Y / 2);
              matrix.fillRect(0, 0, PANEL_RES_X, h, 0);
              matrix.fillRect(0, PANEL_RES_Y - h, PANEL_RES_X, h, 0);
            }
          } else if (activeAnimType == 16) {
            if (p < 0.7) {
              int rx = random(-20, 20);
              drawScreen(activeScreens[currentScreenIdx], rx, 0, &globalTimeInfo, globalTimeOk);
              for (int i = 0; i < 8; i++) {
                matrix.fillRect(random(PANEL_RES_X), random(PANEL_RES_Y), random(20, 80), random(2, 6), 0);
                matrix.fillRect(random(PANEL_RES_X), random(PANEL_RES_Y), random(10, 50), random(1, 3), getColor(0, 255, 255));
                matrix.fillRect(random(PANEL_RES_X), random(PANEL_RES_Y), random(10, 50), random(1, 3), getColor(255, 0, 255));
              }
            } else {
              drawScreen(activeScreens[nextScreenIdx], 0, 0, &globalTimeInfo, globalTimeOk);
            }
          } else if (activeAnimType == 17) {
            int lx = p * (PANEL_RES_X + PANEL_RES_Y) - PANEL_RES_Y;
            if (p < 0.5) {
              drawScreen(activeScreens[currentScreenIdx], 0, 0, &globalTimeInfo, globalTimeOk);
              for (int y = 0; y < PANEL_RES_Y; y++) {
                int x = lx + y;
                matrix.drawLine(0, y, x, y, 0);
              }
            } else {
              drawScreen(activeScreens[nextScreenIdx], 0, 0, &globalTimeInfo, globalTimeOk);
              for (int y = 0; y < PANEL_RES_Y; y++) {
                int x = lx + y;
                matrix.drawLine(x, y, PANEL_RES_X, y, 0);
              }
            }
            matrix.drawLine(lx, 0, lx + PANEL_RES_Y, PANEL_RES_Y, getColor(255, 0, 0));
            matrix.drawLine(lx - 1, 0, lx + PANEL_RES_Y - 1, PANEL_RES_Y, getColor(255, 100, 100));
          } else if (activeAnimType == 18) {
            int sections = 8, secW = PANEL_RES_X / sections;
            if (p < 0.5) {
              drawScreen(activeScreens[currentScreenIdx], 0, 0, &globalTimeInfo, globalTimeOk);
              int bw = (p * 2.0) * secW;
              for (int i = 0; i < sections; i++) { matrix.fillRect(i * secW + (secW - bw) / 2, 0, bw, PANEL_RES_Y, 0); }
            } else {
              drawScreen(activeScreens[nextScreenIdx], 0, 0, &globalTimeInfo, globalTimeOk);
              int bw = ((1.0 - p) * 2.0) * secW;
              for (int i = 0; i < sections; i++) { matrix.fillRect(i * secW + (secW - bw) / 2, 0, bw, PANEL_RES_Y, 0); }
            }
          }
        }
      } else {
        drawScreen(activeScreens[currentScreenIdx], 0, 0, &globalTimeInfo, globalTimeOk);
      }
    }
  }

  if (timerActive) {
    long r = (timerEndTime - ms) / 1000;
    if (r <= 0) {
      timerActive = false;
      timerAlert = true;
      alertStartTime = millis();
      lastAlertSoundTime = 0;
    } else {
      u8g2_fonts.setFont(u8g2_font_cu12_t_cyrillic);
      u8g2_fonts.setForegroundColor(getColor(255, 255, 255));
      u8g2_fonts.setCursor(2, 12);
      u8g2_fonts.printf("%02d:%02d", (int)r / 60, (int)r % 60);
    }
  }

  matrix.show();
}
